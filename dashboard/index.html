Get Latest Reading

import requests

device_id = "your-device-id"
base = "https://sage.evl.uic.edu/api/v1"

response = requests.get(f"{base}/devices/{device_id}/latest")
data = response.json()
print(data["payload"])

Download CSV

import requests

device_id = "your-device-id"
day = "2024-01-22"
base = "https://sage.evl.uic.edu/api/v1"

response = requests.get(
    f"{base}/devices/{device_id}/csv",
    params={"day": day}
)

with open(f"{device_id}_{day}.csv", "w") as f:
    f.write(response.text)

List All Devices

import requests

base = "https://sage.evl.uic.edu/api/v1"

response = requests.get(f"{base}/devices")
data = response.json()
devices = data.get("devices", [])

for device in devices:
    print(device)
