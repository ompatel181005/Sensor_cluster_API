# GET latest reading for a device
import requests

base = "https://sage.evl.uic.edu/api/v1"
device_id = "your-device-id"

r = requests.get(
  f"{base}/devices/{device_id}/latest"
)
print(r.json()["payload"])
# GET historical readings for a date range
import requests

base = "https://sage.evl.uic.edu/api/v1"
device_id = "your-device-id"

r = requests.get(
  f"{base}/devices/{device_id}/readings",
  params={
    "from_date": "2024-01-22",
    "to_date":   "2024-01-22"
  }
)
for reading in r.json():
  print(reading["timestamp"],
        reading["payload"])
# GET CSV export for a specific day
import requests

base = "https://sage.evl.uic.edu/api/v1"
device_id = "your-device-id"
day = "2024-01-22"

r = requests.get(
  f"{base}/devices/{device_id}/csv",
  params={"day": day}
)
with open(f"{device_id}_{day}.csv","w") as f:
  f.write(r.text)
# Route: /ws/devices/{device_id}
# Pushes every new reading in real-time
# Message shape matches /latest response

# ── Python ──
import asyncio, json
import websockets

device_id = "your-device-id"
uri = ("wss://sage.evl.uic.edu"
       f"/ws/devices/{device_id}")

async def listen():
  async with websockets.connect(uri) as ws:
    async for msg in ws:
      d = json.loads(msg)
      # d = {
      #   "id": 123,
      #   "device_id": "jetson-lab-01",
      #   "timestamp": "2024-01-22T14:00:00",
      #   "payload": {
      #     "temperature_c": 22.5,
      #     "humidity_relative_percent": 45.1,
      #     "pressure_hpa": 1013.2,
      #     "gas_resistance_ohms": 45000
      #   }
      # }
      print(d["payload"])

asyncio.run(listen())
